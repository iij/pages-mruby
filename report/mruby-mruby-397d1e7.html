+{
+  mrb_value str;
+
+  str = mrb_str_new(mrb, RSTRING_PTR(str0), RSTRING_LEN(str0));
+  if (strlen(RSTRING_PTR(str)) != RSTRING_LEN(str)) {
+    mrb_raise(mrb, E_ARGUMENT_ERROR, "string contains null byte");
+  }
+  return RSTRING_PTR(str);
