+static void
+change_gen_gc_mode(mrb_state *mrb, mrb_int enable)
+{
+  if (is_generational(mrb) && !enable) {
+    if (is_major_gc(mrb)) {
+      advance_phase(mrb, GC_STATE_NONE);
+    }
+    else {
+      clear_all_old(mrb);

+    }
