 static int
+parse_heredoc_line(parser_state *p)
+{
+  parser_heredoc_info *inf = parsing_heredoc_inf(p);
+  /* assert(inf != NULL); */
+  int c;
+
+  newtok(p);
+  while ((c = nextc(p)) != '\n') {
+    if (c == -1)
+      break;
+    if (inf->type != heredoc_type_quote) {
+      if (c == '\\') {
+	tokadd(p, read_escape(p));
+	inf->line_head = FALSE;
+        continue;
+      }
+      if (c == '#') {
+	c = nextc(p);
+	if (c == '{') {
+	  tokfix(p);
+	  p->lstate = EXPR_BEG;
+	  p->sterm = ' ';
+	  p->cmd_start = TRUE;
+	  yylval.nd = new_str(p, tok(p), toklen(p));
+	  inf->line_head = FALSE;
+	  return tSTRING_PART;
+	}
+	tokadd(p, '#');
+	pushback(p, c);
+	continue;
+      }
+    }
+    tokadd(p, c);
+  }
+  tokadd(p, '\n');
+  tokfix(p);
+  p->lineno++;
+  p->column = 0;
+  int line_head = inf->line_head;
+  inf->line_head = TRUE;
+  if (line_head) {
+    /* check whether end of heredoc */
+    const char *s = tok(p);
+    int len = toklen(p);
+    if (inf->allow_indent) {
+      while (ISSPACE(*s) && len > 0) {
+	++s;
+	--len;
+      }
+    }
+    if ((len-1 == inf->term_len) && (strncmp(s, inf->term, len-1) == 0)) {
